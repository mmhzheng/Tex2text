# 消除注释
%.*\n -> $0

# 消除begin...end关键字
\\begin{abstract}(.*)\\end{abstract} -> $1:s
\\begin{IEEEkeywords}(.*)\\end{IEEEkeywords} -> $1:s

#消除图片
\\begin{figure}.*\\end{figure} -> $0:s

#消除列表
\\begin{itemize}(.*)\\end{itemize} -> $1:s
\\item (.*?) -> $1

# 消除常见的命令
\\IEEEtitleabstractindextext{(.*)} $1:s
\\section{(.*)} : s
\\label{.*?}
~\\cite{.*?}
\\textit{(.*?)}
\\textbf{(.*?)} -> $1:
~\\ref{.*?} -> :

# 替换数学表达式
\$.*?\$ -> EXP:
